#include <stdio.h>
int main(void)
{
	long long int t,n,p,a[1000],i,j,k;
	scanf("%lld",&t);
	while(t--)
	{
		int c=0;
		scanf("%lld %lld",&n,&p);
		for(int i=0;i<n;i++)
			scanf("%lld",&a[i]);
		for(int i=0;i<n;i++)
		{
			if(p%a[i]==0 && p>a[i]) c++;
		}
		if(c==n)
			printf("NO");
		else
		{
			printf("YES");
			long int max=0;
			long long int s=0,x=1;
			for(i=0;i<n;i++)
				if(max<a[i])
					max=a[i];
			while(s<p)
			{
				s=s+max;
				x++;
			}
			for(i=0;i<n;i++)
			{
				if(a[i]==max)
				{
				        printf(" %lld",x-1);
				        continue;
				}
				printf(" 0");
			}
		}
		printf("\n");
	}
	return 0;
}
