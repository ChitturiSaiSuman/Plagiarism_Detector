#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>
#include <string.h>
int palindrome(int a[],int p)
{
    int i,n;
    n=p;
    for(i=0;i<n/2;i++)
    {
        if(a[i]!=a[n-i-1])
        {
	    break;
	}
    }
    if(i==n/2)
	return 1;
    else
	return 0;
}
int min(int a,int b)
{
    if(a>b)
	return b;
    else
	return a;
}
int main()
{
    int a1[100],a2[100],a[50][50],o,min,i,j,m,n,q,k,t,h,s;
    scanf("%d",&t);
    for(o=0;o<t;o++)
    {
	int c=0;
	scanf("%d %d",&m,&n);
	min=min(m,n);
	for(i=0;i<m;i++)
	{
	    for(j=0;j<n;j++)
	    {
	        scanf("%d",&a[i][j]);
	    }
	}
	for(h=3;h<=min;h+=2)
	{
	    for(q=0;q<(m-h+1);q++)
	    {
	        for(k=0;k<(n-h+1);k++)
	        {
	            int p=0;
	            while(p<h)
	            {
	                a1[p]=a[p+q][k+(h/2)];
	                a2[p]=a[(h/2)+q][k+p];
	                p++;
	            	if(palindrome(a1,p)==1 && palindrome(a2,p)==1)
		        {
	               	    c++;
	            	}
	            }
	        }
	    }
	}
	printf("%d",c+(m*n));
    }
    return 0;
}
